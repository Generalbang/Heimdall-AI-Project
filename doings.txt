// src/BlogForm.js
import React, { useState } from "react";

const BlogForm = ({ addPost }) => {
  const [title, setTitle] = useState("");
  const [content, setContent] = useState("");

  const handleSubmit = async (e) => {
    e.preventDefault();

    // Ensure both fields are filled
    if (!title || !content) {
      alert("Title and content are required!");
      return;
    }

    // POST request to backend API
    const response = await fetch("http://localhost:5000/posts", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
      },
      body: JSON.stringify({ title, content }),
    });

    const data = await response.json();
    if (response.ok) {
      addPost(data); // Update the list of posts after adding a new one
      setTitle(""); // Clear form fields
      setContent("");
    } else {
      alert("Error: " + data.message);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label htmlFor="title">Title</label>
        <input
          type="text"
          id="title"
          value={title}
          onChange={(e) => setTitle(e.target.value)}
          placeholder="Blog Title"
        />
      </div>
      <div>
        <label htmlFor="content">Content</label>
        <textarea
          id="content"
          value={content}
          onChange={(e) => setContent(e.target.value)}
          placeholder="Blog Content"
        />
      </div>
      <button type="submit">Submit Post</button>
    </form>
  );
};

// src/BlogList.js
import React, { useState, useEffect } from "react";

const BlogList = ({ posts, setPosts }) => {
  const deletePost = async (id) => {
    if (window.confirm("Are you sure you want to delete this post?")) {
      const response = await fetch(`http://localhost:5000/posts/${id}`, {
        method: "DELETE",
      });

      if (response.ok) {
        // Remove the post from the list if successful
        setPosts((prevPosts) => prevPosts.filter((post) => post._id !== id));
      } else {
        alert("Error deleting post");
      }
    }
  };

  return (
    <div>
      {posts.map((post) => (
        <div key={post._id}>
          <h3>{post.title}</h3>
          <p>{new Date(post.timestamp).toLocaleString()}</p>
          <button onClick={() => deletePost(post._id)}>Delete</button>
        </div>
      ))}
    </div>
  );
};

export default BlogList;
Step 3: App Component (Integrating Form & List)
Now, letâ€™s connect everything inside App.js. It will manage the state for the posts and provide the necessary props for BlogForm and BlogList.

// src/App.js
import React, { useState, useEffect } from "react";
import BlogForm from "./BlogForm";
import BlogList from "./BlogList";

const App = () => {
  const [posts, setPosts] = useState([]);

  // Fetch posts on component mount
  useEffect(() => {
    const fetchPosts = async () => {
      const response = await fetch("http://localhost:5000/posts");
      const data = await response.json();
      setPosts(data);
    };
    fetchPosts();
  }, []);

  // Add a new post to the list
  const addPost = (newPost) => {
    setPosts((prevPosts) => [newPost, ...prevPosts]);
  };

  return (
    <div>
      <h1>Blog Platform</h1>
      <BlogForm addPost={addPost} />
      <BlogList posts={posts} setPosts={setPosts} />
    </div>
  );
};

export default App;
